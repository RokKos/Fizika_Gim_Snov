\begin{align*}
	p &= S_0 e^{-\beta t}\\
	\beta &\dots \text{faktor dušenja[$s^{-1}$]}\\
	\frac{1}{\beta} &\dots \text{Je čas v katerem pade \textbf{amplituda} na \textbf{$\frac{1}{e}$}}\\
	\frac{1}{e} &= 0,368\\
\end{align*}
